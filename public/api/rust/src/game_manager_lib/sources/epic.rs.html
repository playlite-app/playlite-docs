1//! Source para importar jogos da Epic Games
2//!
3//! Detecta jogos instalados lendo os arquivos de manifesto `.item` do Epic Games Launcher.
4//! Importa a biblioteca completa via OAuth2 (login na conta Epic).
5//!
6//! **Observações:**
7//! - **Windows:** os manifestos estão em `C:\ProgramData\Epic\EpicGamesLauncher\Data\Manifests`.
8//! - **Linux (Wine):** o mesmo caminho é resolvido dentro do Wine prefix em
9//!   `<prefix>/drive_c/ProgramData/Epic/EpicGamesLauncher/Data/Manifests`.
10//! - Cada arquivo `.item` é um JSON com nome, caminho de instalação e executável do jogo.
11

34/// Estrutura mínima do JSON dos arquivos `.item`
35#[derive(Debug, Deserialize)]
36#[serde(rename_all = "PascalCase")]
37struct EpicManifest {
38    display_name: Option<String>,
39    install_location: Option<String>,
40    launch_executable: Option<String>,
41    app_name: Option<String>,
42}

44/// Source responsável por importar jogos instalados via Epic Games
45pub struct EpicSource {
46    app_handle: AppHandle,
47    #[allow(dead_code)]
48    wine_prefix: Option<PathBuf>, // Wine prefix utilizado no Linux para localizar os manifestos do Epic. Ignorado no Windows.
49    config: OAuthProviderConfig,
50}

52#[derive(Debug, Deserialize, Clone)]
53struct EpicLibraryItem {
54    namespace: String,
55    #[serde(rename = "catalogItemId")]
56    catalog_item_id: String,
57    #[serde(rename = "sandboxName")]
58    sandbox_name: Option<String>,
59}
60
61#[derive(Debug, Deserialize)]
62struct EpicLibraryResponse {
63    records: Vec<EpicLibraryItem>,
64    #[serde(rename = "responseMetadata", default)]
65    response_metadata: Option<EpicLibraryMetadata>,
66}
67
68#[derive(Debug, Deserialize)]
69struct EpicLibraryMetadata {
70    #[serde(rename = "nextCursor", default)]
71    next_cursor: Option<String>,
72}

76impl EpicSource {
77    pub fn new(app_handle: AppHandle, wine_prefix: Option<PathBuf>) -> Self {
78        let config = OAuthProviderConfig {
79            provider_id: "epic",
80            client_id: EPIC_OAUTH_CLIENT_ID.to_string(),
81            client_secret: Some(EPIC_OAUTH_CLIENT_SECRET.to_string()),
82            authorize_endpoint: EPIC_LOGIN_URL.to_string(),
83            token_endpoint: EPIC_TOKEN_ENDPOINT.to_string(),
84            redirect_uri: EPIC_REDIRECT_PREFIX.to_string(),
85            scopes: vec![],
86            uses_pkce: false,
87            extra_params: vec![],
88            token_request_method: TokenRequestMethod::Post,
89            token_auth_method: TokenAuthMethod::BasicHeader,
90        };
91
92        Self {
93            app_handle,
94            wine_prefix,
95            config,
96        }
97    }
98
99    /// Resolve o diretório de manifestos do Epic Games Launcher.
100    ///
101    /// - **Windows:** `C:\ProgramData\Epic\EpicGamesLauncher\Data\Manifests`
102    /// - **Linux:** `<wine_prefix>/drive_c/ProgramData/Epic/EpicGamesLauncher/Data/Manifests`
103    fn resolve_manifest_dir(&self) -> Option<PathBuf> {
104        #[cfg(target_os = "windows")]
105        {
106            use crate::constants::EPIC_MANIFEST_PATH_WINDOWS;
107            let path = PathBuf::from(EPIC_MANIFEST_PATH_WINDOWS);
108            if path.exists() {
109                return Some(path);
110            }
111        }
112
113        #[cfg(target_os = "linux")]
114        {
115            if let Some(prefix) = &self.wine_prefix {
116                let path = prefix
117                    .join("drive_c")
118                    .join("ProgramData")
119                    .join("Epic")
120                    .join("EpicGamesLauncher")
121                    .join("Data")
122                    .join("Manifests");
123                if path.exists() {
124                    return Some(path);
125                }
126            }
127        }
128
129        None
130    }
131
132    /// Importa todos os jogos instalados detectados nos manifestos locais
133    pub async fn import_installed(&self) -> Result<Vec<SourceGame>, AppError> {
134        let manifest_dir = match self.resolve_manifest_dir() {
135            Some(dir) => dir,
136            None => return Ok(vec![]), // Epic não instalada ou sem jogos
137        };
138
139        let mut games = Vec::new();
140
141        for entry in fs::read_dir(&manifest_dir)? {
142            let entry = entry?;
143            let path = entry.path();
144
145            if !is_item_file(&path) {
146                continue;
147            }
148
149            match Self::parse_manifest(&path) {
150                Ok(game) => games.push(game),
151                Err(err) => {
152                    eprintln!("Erro ao processar manifest {:?}: {}", path, err);
153                    continue;
154                }
155            }
156        }
157
158        Ok(games)
159    }
160
161    fn parse_manifest(path: &Path) -> Result<SourceGame, AppError> {
162        let content = fs::read_to_string(path)?;
163
164        let manifest: EpicManifest = serde_json::from_str(&content)?;
165
166        let name = manifest
167            .display_name
168            .unwrap_or_else(|| "Unknown".to_string());
169
170        let install_location = manifest
171            .install_location
172            .ok_or_else(|| AppError::EpicMissingField("InstallLocation".into()))?;
173
174        let launch_executable = manifest
175            .launch_executable
176            .ok_or_else(|| AppError::EpicMissingField("LaunchExecutable".into()))?;
177
178        let app_name = manifest
179            .app_name
180            .ok_or_else(|| AppError::EpicMissingField("AppName".into()))?;
181
182        let full_executable_path = Path::new(&install_location)
183            .join(&launch_executable)
184            .to_string_lossy()
185            .to_string();
186
187        Ok(SourceGame {
188            platform: "Epic".to_string(),
189            platform_game_id: app_name,
190            name: Some(name),
191            installed: true,
192            executable_path: Some(full_executable_path),
193            install_path: Some(install_location),
194            playtime_minutes: None,
195            last_played: None,
196        })
197    }
198}

202impl EpicSource {
203    /// Importa a biblioteca completa de jogos possuídos na conta Epic (requer login prévio).
204    pub async fn fetch_library_detailed(&self) -> Result<Vec<SourceGame>, AppError> {
205        let access_token = self.ensure_valid_token().await?;
206        let items = fetch_all_library_items(&access_token).await?;
207
208        let mut seen_namespaces = std::collections::HashSet::new();
209        let mut games = Vec::new();
210
211        for item in items {
212            if item.namespace == "ue" {
213                continue; // addons/assets de Unreal Engine, não são jogos
214            }
215
216            // Múltiplos catalogItemId podem compartilhar o mesmo namespace (builds/plataformas
217            // diferentes do mesmo título) — mantém só a primeira ocorrência por jogo.
218            if !seen_namespaces.insert(item.namespace.clone()) {
219                continue;
220            }
221
222            let name = item
223                .sandbox_name
224                .filter(|n| !n.trim().is_empty())
225                .unwrap_or_else(|| item.catalog_item_id.clone());
226
227            games.push(SourceGame {
228                platform: "Epic".to_string(),
229                platform_game_id: item.namespace,
230                name: Some(name),
231                installed: false,
232                executable_path: None,
233                install_path: None,
234                playtime_minutes: None,
235                last_played: None,
236            });
237        }
238
239        Ok(games)
240    }
241}

250#[async_trait]
251impl OAuthGameSource for EpicSource {
252    fn oauth_config(&self) -> &OAuthProviderConfig {
253        &self.config
254    }
255
256    fn app_handle(&self) -> &AppHandle {
257        &self.app_handle
258    }
259
260    /// A Epic não devolve `code` na query do redirect (como o GOG) — o `authorizationCode`
261    /// vem como JSON no corpo da própria página de redirect. Um script injetado via
262    /// `on_page_load` lê esse corpo e navega para uma pseudo-URL própria
263    /// (`playlite://epic-auth-code?data=...`), que o `on_navigation` consegue interceptar
264    /// (ele só recebe a URL da navegação, nunca o conteúdo da página).
265    async fn login(&self) -> Result<(), AppError> {
266        let config = self.oauth_config();
267        let login_url = url::Url::parse(&config.authorize_endpoint)
268            .map_err(|e| AppError::OAuthConfigError(format!("URL de login Epic inválida: {e}")))?;
269
270        let (tx, rx) = mpsc::channel::<Result<String, String>>();
271        let tx_nav = tx.clone();
272
273        let window = WebviewWindowBuilder::new(
274            &self.app_handle,
275            "epic_oauth_login",
276            WebviewUrl::External(login_url),
277        )
278            .title("Login Epic Games")
279            .inner_size(480.0, 720.0)
280            .on_navigation(move |url| {
281                let url_str = url.as_str();
282                if url_str.starts_with(EPIC_PSEUDO_REDIRECT_SCHEME) {
283                    let data = url
284                        .query_pairs()
285                        .find(|(k, _)| k == "data")
286                        .map(|(_, v)| v.to_string());
287                    let result =
288                        data.ok_or_else(|| "Payload 'data' ausente no callback Epic".to_string());
289                    let _ = tx_nav.send(result);
290                    return false;
291                }
292                true
293            })
294            .on_page_load(move |window, payload| {
295                if payload.url().as_str().starts_with(EPIC_REDIRECT_PREFIX) {
296                    let script = format!(
297                        r#"(function() {{
298                        try {{
299                            var data = document.body.innerText || document.body.textContent || "";
300                            window.location.href = "{scheme}?data=" + encodeURIComponent(data);
301                        }} catch (e) {{
302                            window.location.href = "{scheme}?data=" + encodeURIComponent(JSON.stringify({{error: String(e)}}));
303                        }}
304                    }})();"#,
305                        scheme = EPIC_PSEUDO_REDIRECT_SCHEME
306                    );
307
308                    let _ = window.eval(&script);
309                }
310            })
311            .build()
312            .map_err(|e| {
313                AppError::OAuthConfigError(format!("Falha ao abrir janela de login Epic: {e}"))
314            })?;
315
316        let raw_json = tokio::task::spawn_blocking(move || {
317            rx.recv_timeout(Duration::from_secs(OAUTH_CALLBACK_TIMEOUT_SECS))
318        })
319        .await
320        .map_err(|e| AppError::OAuthConfigError(format!("Task de callback falhou: {e}")))?
321        .map_err(|_| AppError::OAuthConfigError("Tempo limite de login excedido".to_string()))?
322        .map_err(AppError::OAuthConfigError)?;
323
324        let _ = window.close();
325
326        #[derive(Deserialize)]
327        struct EpicAuthRedirect {
328            #[serde(rename = "authorizationCode")]
329            authorization_code: Option<String>,
330        }
331
332        let parsed: EpicAuthRedirect = serde_json::from_str(&raw_json).map_err(|e| {
333            AppError::OAuthConfigError(format!("Resposta de login Epic inesperada: {e}"))
334        })?;
335
336        let code = parsed.authorization_code.ok_or_else(|| {
337            AppError::OAuthConfigError(
338                "Epic não retornou 'authorizationCode' — login cancelado ou falhou".to_string(),
339            )
340        })?;
341
342        let token_response = exchange_code_for_token(config, &code, None).await?;
343        save_oauth_token(&self.app_handle, config.provider_id, &token_response.into())?;
344
345        Ok(())
346    }
347}

351/// Busca a biblioteca completa da Epic Games
352/// Esse endpoint não expôs nenhum campo de paginação (`responseMetadata`/cursor).
353/// Na prática retorna a lista inteira numa resposta só.
354async fn fetch_all_library_items(access_token: &str) -> Result<Vec<EpicLibraryItem>, AppError> {
355    let mut all = Vec::new();
356    let mut cursor: Option<String> = None;
357
358    loop {
359        let mut request = HTTP_CLIENT
360            .get(EPIC_LIBRARY_ENDPOINT)
361            .bearer_auth(access_token)
362            .query(&[("includeMetadata", "true")]);
363
364        if let Some(c) = &cursor {
365            request = request.query(&[("cursor", c)]);
366        }
367
368        let response = request.send().await?;
369        let status = response.status();
370        let body = response.text().await.unwrap_or_default();
371
372        if !status.is_success() {
373            return Err(AppError::NetworkError(format!(
374                "Epic library retornou HTTP {status}: {body}"
375            )));
376        }
377
378        let parsed: EpicLibraryResponse = serde_json::from_str(&body).map_err(|e| {
379            AppError::ParseError(format!(
380                "Falha ao parsear biblioteca Epic: {e} — corpo: {body}"
381            ))
382        })?;
383
384        let record_count = parsed.records.len();
385        all.extend(parsed.records);
386
387        let next_cursor = parsed.response_metadata.and_then(|m| m.next_cursor);
388        log::debug!(
389            "Epic library: página com {record_count} itens (total acumulado: {})",
390            all.len()
391        );
392
393        match next_cursor {
394            Some(c) if !c.is_empty() => cursor = Some(c),
395            _ => break,
396        }
397    }
398
399    Ok(all)
400}

402/// Cruza a biblioteca completa (OAuth) com os jogos detectados localmente via manifesto,
403/// marcando como instalados os que baterem por nome (case-insensitive) — já que o `AppName`
404/// do manifesto local e o `catalogItemId` da API não são o mesmo ID. Jogos instalados que,
405/// por algum motivo, não aparecerem na biblioteca são mantidos como entradas à parte.
406pub fn merge_local_install_status(
407    library_games: &mut Vec<SourceGame>,
408    local_games: Vec<SourceGame>,
409) {
410    for local in local_games {
411        let local_name = local
412            .name
413            .as_deref()
414            .unwrap_or_default()
415            .trim()
416            .to_lowercase();
417
418        let matched = library_games
419            .iter_mut()
420            .find(|g| g.name.as_deref().unwrap_or_default().trim().to_lowercase() == local_name);
421
422        match matched {
423            Some(g) => {
424                g.installed = true;
425                g.executable_path = local.executable_path;
426                g.install_path = local.install_path;
427            }
428            None => library_games.push(local),
429        }
430    }
431}

433fn is_item_file(path: &Path) -> bool {
434    path.extension()
435        .and_then(|s| s.to_str())
436        .map(|ext| ext.eq_ignore_ascii_case("item"))
437        .unwrap_or(false)
