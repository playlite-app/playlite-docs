1//! Source para importar jogos da Epic Games
2//!
3//! Detecta jogos instalados lendo os arquivos de manifesto `.item` do Epic Games Launcher.
4//! Importa a biblioteca completa via OAuth2 (login na conta Epic).
5//!
6//! **Observações:**
7//! - **Windows:** os manifestos estão em `C:\ProgramData\Epic\EpicGamesLauncher\Data\Manifests`.
8//! - **Linux (Wine):** o mesmo caminho é resolvido dentro do Wine prefix em
9//!   `<prefix>/drive_c/ProgramData/Epic/EpicGamesLauncher/Data/Manifests`.
10//! - Cada arquivo `.item` é um JSON com nome, caminho de instalação e executável do jogo.

94impl EpicSource {
95    pub fn new(app_handle: AppHandle, wine_prefix: Option<PathBuf>) -> Self {
96        let config = OAuthProviderConfig {
97            provider_id: "epic",
98            client_id: EPIC_OAUTH_CLIENT_ID.to_string(),
99            client_secret: Some(EPIC_OAUTH_CLIENT_SECRET.to_string()),
100            authorize_endpoint: EPIC_LOGIN_URL.to_string(),
101            token_endpoint: EPIC_TOKEN_ENDPOINT.to_string(),
102            redirect_uri: EPIC_REDIRECT_PREFIX.to_string(),
103            scopes: vec![],
104            uses_pkce: false,
105            extra_params: vec![],
106            token_request_method: TokenRequestMethod::Post,
107            token_auth_method: TokenAuthMethod::BasicHeader,
108        };
109
110        Self {
111            app_handle,
112            wine_prefix,
113            config,
114        }
115    }
116
117    /// Resolve o diretório de manifestos do Epic Games Launcher.
118    ///
119    /// - **Windows:** `C:\ProgramData\Epic\EpicGamesLauncher\Data\Manifests`
120    /// - **Linux:** `<wine_prefix>/drive_c/ProgramData/Epic/EpicGamesLauncher/Data/Manifests`
121    fn resolve_manifest_dir(&self) -> Option<PathBuf> {
122        #[cfg(target_os = "windows")]
123        {
124            use crate::constants::EPIC_MANIFEST_PATH_WINDOWS;
125            let path = PathBuf::from(EPIC_MANIFEST_PATH_WINDOWS);
126            if path.exists() {
127                return Some(path);
128            }
129        }
130
131        #[cfg(target_os = "linux")]
132        {
133            if let Some(prefix) = &self.wine_prefix {
134                let path = prefix
135                    .join("drive_c")
136                    .join("ProgramData")
137                    .join("Epic")
138                    .join("EpicGamesLauncher")
139                    .join("Data")
140                    .join("Manifests");
141                if path.exists() {
142                    return Some(path);
143                }
144            }
145        }
146
147        None
148    }
149
150    /// Importa todos os jogos instalados detectados nos manifestos locais
151    pub async fn import_installed(&self) -> Result<Vec<SourceGame>, AppError> {
152        let manifest_dir = match self.resolve_manifest_dir() {
153            Some(dir) => dir,
154            None => return Ok(vec![]), // Epic não instalada ou sem jogos
155        };
156
157        let mut games = Vec::new();
158
159        for entry in fs::read_dir(&manifest_dir)? {
160            let entry = entry?;
161            let path = entry.path();
162
163            if !is_item_file(&path) {
164                continue;
165            }
166
167            match Self::parse_manifest(&path) {
168                Ok(game) => games.push(game),
169                Err(err) => {
170                    log::warn!("Erro ao processar manifest {:?}: {}", path, err);
171                    continue;
172                }
173            }
174        }
175
176        Ok(games)
177    }
178
179    fn parse_manifest(path: &Path) -> Result<SourceGame, AppError> {
180        let content = fs::read_to_string(path)?;
181
182        let manifest: EpicManifest = serde_json::from_str(&content)?;
183
184        let name = manifest
185            .display_name
186            .unwrap_or_else(|| "Unknown".to_string());
187
188        let install_location = manifest
189            .install_location
190            .ok_or_else(|| AppError::EpicMissingField("InstallLocation".into()))?;
191
192        let launch_executable = manifest
193            .launch_executable
194            .ok_or_else(|| AppError::EpicMissingField("LaunchExecutable".into()))?;
195
196        let app_name = manifest
197            .app_name
198            .ok_or_else(|| AppError::EpicMissingField("AppName".into()))?;
199
200        let full_executable_path = Path::new(&install_location)
201            .join(&launch_executable)
202            .to_string_lossy()
203            .to_string();
204
205        Ok(SourceGame {
206            platform: "Epic".to_string(),
207            platform_game_id: app_name,
208            name: Some(name),
209            installed: true,
210            executable_path: Some(full_executable_path),
211            install_path: Some(install_location),
212            playtime_minutes: None,
213            last_played: None,
214        })
215    }
216}

220impl EpicSource {
221    /// Importa a biblioteca completa de jogos possuídos na conta Epic (requer login prévio).
222    pub async fn fetch_library_detailed(&self) -> Result<Vec<SourceGame>, AppError> {
223        let access_token = self.ensure_valid_token().await?;
224        let items = fetch_all_library_items(&access_token).await?;
225
226        let mut by_namespace: HashMap<String, Vec<String>> = HashMap::new();
227        for item in &items {
228            if item.namespace == "ue" {
229                continue;
230            }
231            by_namespace
232                .entry(item.namespace.clone())
233                .or_default()
234                .push(item.catalog_item_id.clone());
235        }
236
237        let titles = fetch_all_catalog_titles(&access_token, &by_namespace).await?;
238
239        // Escolhe, por namespace, o catalogItemId marcado como "games" no catálogo.
240        let mut chosen_per_namespace: HashMap<String, (String, String)> = HashMap::new(); // namespace -> (catalog_item_id, title)
241
242        for item in &items {
243            if item.namespace == "ue" {
244                continue;
245            }
246
247            let Some((title, is_game)) =
248                titles.get(&(item.namespace.clone(), item.catalog_item_id.clone()))
249            else {
250                continue; // catálogo não resolveu esse item; ignora, não usa como fallback de namespace
251            };
252
253            if *is_game {
254                chosen_per_namespace
255                    .entry(item.namespace.clone())
256                    .or_insert_with(|| (item.catalog_item_id.clone(), title.clone()));
257            }
258        }
259
260        let games = chosen_per_namespace
261            .into_iter()
262            .map(|(namespace, (_, title))| SourceGame {
263                platform: "Epic".to_string(),
264                platform_game_id: namespace,
265                name: Some(title),
266                installed: false,
267                executable_path: None,
268                install_path: None,
269                playtime_minutes: None,
270                last_played: None,
271            })
272            .collect();
273
274        Ok(games)
275    }
276}

285#[async_trait]
286impl OAuthGameSource for EpicSource {
287    fn oauth_config(&self) -> &OAuthProviderConfig {
288        &self.config
289    }
290
291    fn app_handle(&self) -> &AppHandle {
292        &self.app_handle
293    }
294
295    /// A Epic não devolve `code` na query do redirect (como o GOG) — o `authorizationCode`
296    /// vem como JSON no corpo da própria página de redirect. Um script injetado via
297    /// `on_page_load` lê esse corpo e navega para uma pseudo-URL própria
298    /// (`playlite://epic-auth-code?data=...`), que o `on_navigation` consegue interceptar
299    /// (ele só recebe a URL da navegação, nunca o conteúdo da página).
300    async fn login(&self) -> Result<(), AppError> {
301        let config = self.oauth_config();
302        let login_url = url::Url::parse(&config.authorize_endpoint)
303            .map_err(|e| AppError::OAuthConfigError(format!("URL de login Epic inválida: {e}")))?;
304
305        let (tx, rx) = mpsc::channel::<Result<String, String>>();
306        let tx_nav = tx.clone();
307
308        let window = WebviewWindowBuilder::new(
309            &self.app_handle,
310            "epic_oauth_login",
311            WebviewUrl::External(login_url),
312        )
313            .title("Login Epic Games")
314            .inner_size(480.0, 720.0)
315            .on_navigation(move |url| {
316                let url_str = url.as_str();
317                if url_str.starts_with(EPIC_PSEUDO_REDIRECT_SCHEME) {
318                    let data = url
319                        .query_pairs()
320                        .find(|(k, _)| k == "data")
321                        .map(|(_, v)| v.to_string());
322                    let result =
323                        data.ok_or_else(|| "Payload 'data' ausente no callback Epic".to_string());
324                    let _ = tx_nav.send(result);
325                    return false;
326                }
327                true
328            })
329            .on_page_load(move |window, payload| {
330                if payload.url().as_str().starts_with(EPIC_REDIRECT_PREFIX) {
331                    let script = format!(
332                        r#"(function() {{
333                        try {{
334                            var data = document.body.innerText || document.body.textContent || "";
335                            window.location.href = "{scheme}?data=" + encodeURIComponent(data);
336                        }} catch (e) {{
337                            window.location.href = "{scheme}?data=" + encodeURIComponent(JSON.stringify({{error: String(e)}}));
338                        }}
339                    }})();"#,
340                        scheme = EPIC_PSEUDO_REDIRECT_SCHEME
341                    );
342
343                    let _ = window.eval(&script);
344                }
345            })
346            .build()
347            .map_err(|e| {
348                AppError::OAuthConfigError(format!("Falha ao abrir janela de login Epic: {e}"))
349            })?;
350
351        let raw_json = tokio::task::spawn_blocking(move || {
352            rx.recv_timeout(Duration::from_secs(OAUTH_CALLBACK_TIMEOUT_SECS))
353        })
354        .await
355        .map_err(|e| AppError::OAuthConfigError(format!("Task de callback falhou: {e}")))?
356        .map_err(|_| AppError::OAuthConfigError("Tempo limite de login excedido".to_string()))?
357        .map_err(AppError::OAuthConfigError)?;
358
359        let _ = window.close();
360
361        #[derive(Deserialize)]
362        struct EpicAuthRedirect {
363            #[serde(rename = "authorizationCode")]
364            authorization_code: Option<String>,
365        }
366
367        let parsed: EpicAuthRedirect = serde_json::from_str(&raw_json).map_err(|e| {
368            AppError::OAuthConfigError(format!("Resposta de login Epic inesperada: {e}"))
369        })?;
370
371        let code = parsed.authorization_code.ok_or_else(|| {
372            AppError::OAuthConfigError(
373                "Epic não retornou 'authorizationCode' — login cancelado ou falhou".to_string(),
374            )
375        })?;
376
377        let token_response = exchange_code_for_token(config, &code, None).await?;
378        save_oauth_token(&self.app_handle, config.provider_id, &token_response.into())?;
379
380        Ok(())
381    }
382}

386/// Busca a biblioteca completa da Epic Games, paginando via cursor.
387///
388/// **Importante:** `includeMetadata` precisa ser `"true"` — com `"false"`, a API omite
389/// `responseMetadata`/`nextCursor` da resposta e retorna só a primeira página sem sinalizar
390/// que há mais páginas disponíveis.
391async fn fetch_all_library_items(access_token: &str) -> Result<Vec<EpicLibraryItem>, AppError> {
392    let mut all = Vec::new();
393    let mut cursor: Option<String> = None;
394
395    loop {
396        let mut request = HTTP_CLIENT
397            .get(EPIC_LIBRARY_ENDPOINT)
398            .bearer_auth(access_token)
399            .query(&[("includeMetadata", "true")]);
400
401        if let Some(c) = &cursor {
402            request = request.query(&[("cursor", c)]);
403        }
404
405        let response = request.send().await?;
406        let status = response.status();
407        let body = response.text().await.unwrap_or_default();
408
409        if !status.is_success() {
410            return Err(AppError::NetworkError(format!(
411                "Epic library retornou HTTP {status}: {body}"
412            )));
413        }
414
415        let parsed: EpicLibraryResponse = serde_json::from_str(&body).map_err(|e| {
416            AppError::ParseError(format!(
417                "Falha ao parsear biblioteca Epic: {e} — corpo: {body}"
418            ))
419        })?;
420
421        let record_count = parsed.records.len();
422        all.extend(parsed.records);
423
424        let next_cursor = parsed.response_metadata.and_then(|m| m.next_cursor);
425        log::debug!(
426            "Epic library: página com {record_count} itens (total acumulado: {})",
427            all.len()
428        );
429
430        match next_cursor {
431            Some(c) if !c.is_empty() => cursor = Some(c),
432            _ => break,
433        }
434    }
435
436    Ok(all)
437}

439/// Resolve os títulos reais dos itens de um namespace via catálogo.
440async fn fetch_catalog_titles(
441    access_token: &str,
442    namespace: &str,
443    catalog_item_ids: &[String],
444) -> Result<HashMap<String, (String, bool)>, AppError> {
445    if catalog_item_ids.is_empty() {
446        return Ok(HashMap::new());
447    }
448
449    let url = format!("{EPIC_CATALOG_BULK_ENDPOINT}/{namespace}/bulk/items");
450
451    let mut query: Vec<(&str, &str)> = catalog_item_ids
452        .iter()
453        .map(|id| ("id", id.as_str()))
454        .collect();
455    query.push(("includeDLCDetails", "false"));
456    query.push(("includeMainGameDetails", "false"));
457
458    let response = HTTP_CLIENT
459        .get(&url)
460        .bearer_auth(access_token)
461        .query(&query)
462        .send()
463        .await?;
464
465    let status = response.status();
466    let body = response.text().await.unwrap_or_default();
467
468    if !status.is_success() {
469        log::warn!("Epic catalog (namespace={namespace}) retornou HTTP {status}: {body}");
470        return Ok(HashMap::new());
471    }
472
473    let parsed: HashMap<String, EpicCatalogItem> = serde_json::from_str(&body).map_err(|e| {
474        AppError::ParseError(format!(
475            "Falha ao parsear catálogo Epic: {e} — corpo: {body}"
476        ))
477    })?;
478
479    Ok(parsed
480        .into_iter()
481        .map(|(id, item)| {
482            let is_game = item.is_game();
483            (id, (item.title, is_game))
484        })
485        .collect())
486}

488/// Resolve os títulos de todos os namespaces em paralelo (até 8 chamadas simultâneas) em vez de sequencialmente — reduz o tempo de import.
489async fn fetch_all_catalog_titles(
490    access_token: &str,
491    by_namespace: &HashMap<String, Vec<String>>,
492) -> Result<HashMap<(String, String), (String, bool)>, AppError> {
493    // Clonar o token de acesso para que cada tarefa assíncrona possua seu próprio dono do token e não dependa de um borrow com lifetime restrito.
494    let access_owned = access_token.to_string();
495
496    let jobs: Vec<(String, Vec<String>)> = by_namespace
497        .iter()
498        .map(|(namespace, ids)| (namespace.clone(), ids.clone()))
499        .collect();
500
501    let results: Vec<Result<(String, HashMap<String, (String, bool)>), AppError>> =
502        stream::iter(jobs)
503            .map(move |(namespace, ids)| {
504                let access = access_owned.clone();
505                async move {
506                    let resolved = fetch_catalog_titles(&access, &namespace, &ids).await?;
507                    Ok::<_, AppError>((namespace, resolved))
508                }
509            })
510            .buffer_unordered(8)
511            .collect()
512            .await;
513
514    let mut titles = HashMap::new();
515    for result in results {
516        let (namespace, resolved) = result?;
517        for (catalog_item_id, (title, is_game)) in resolved {
518            titles.insert((namespace.clone(), catalog_item_id), (title, is_game));
519        }
520    }
521
522    Ok(titles)
523}

525/// Cruza a biblioteca completa (OAuth) com os jogos detectados localmente via manifesto,
526/// marcando como instalados os que baterem por nome (case-insensitive) — já que o `AppName`
527/// do manifesto local e o `catalogItemId` da API não são o mesmo ID. Jogos instalados que,
528/// por algum motivo, não aparecerem na biblioteca são mantidos como entradas à parte.
529pub fn merge_local_install_status(
530    library_games: &mut Vec<SourceGame>,
531    local_games: Vec<SourceGame>,
532) {
533    for local in local_games {
534        let local_name = local
535            .name
536            .as_deref()
537            .unwrap_or_default()
538            .trim()
539            .to_lowercase();
540
541        let matched = library_games
542            .iter_mut()
543            .find(|g| g.name.as_deref().unwrap_or_default().trim().to_lowercase() == local_name);
544
545        match matched {
546            Some(g) => {
547                g.installed = true;
548                g.executable_path = local.executable_path;
549                g.install_path = local.install_path;
550            }
551            None => library_games.push(local),
552        }
553    }
554}

556fn is_item_file(path: &Path) -> bool {
557    path.extension()
558        .and_then(|s| s.to_str())
559        .map(|ext| ext.eq_ignore_ascii_case("item"))
560        .unwrap_or(false)
