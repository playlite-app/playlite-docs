1//! Source para importar a biblioteca da Amazon Games e Scan local de instalados via Amazon Games App.
2//!
3//! Diferente de GOG/Epic, a Amazon não usa OAuth2 padrão — usa um esquema de "registro de
4//! dispositivo" (o mesmo mecanismo usado por apps Kindle/Alexa), baseado em OpenID 2.0 para
5//! o login e um endpoint próprio (`/auth/register`) para trocar o código por tokens.
6//!
7//! Para detectar jogos instalados lê o SQLite mantido pelo próprio launcher em
8//! `%LOCALAPPDATA%\Amazon Games\Data\Games\Sql\GameInstallInfo.sqlite`.
9

127    /// Fluxo de login: gera identidade de dispositivo local (PKCE + serial arbitrário),
128    /// abre a tela de login da Amazon, intercepta o redirect final e troca o código
129    /// obtido por tokens via `register_device`.
130    ///
131    /// **Observação:** o esquema OpenID 2.0 usado aqui não tem parâmetro `state` —
132    /// diferente de GOG/Epic, não há validação de CSRF nesse retorno porque a própria
133    /// Amazon não devolve nada equivalente nesse fluxo.
134    pub async fn login(&self) -> Result<(), AppError> {
135        let pkce = PkceChallenge::generate();
136        let serial = generate_device_serial();
137        let client_id_hex = generate_client_id(&serial);
138
139        let auth_url = build_amazon_auth_url(&client_id_hex, &pkce.challenge)?;
140
141        let (tx, rx) = mpsc::channel::<Result<String, String>>();
142
143        let window = WebviewWindowBuilder::new(
144            &self.app_handle,
145            "amazon_oauth_login",
146            WebviewUrl::External(auth_url),
147        )
148        .title("Login Amazon Games")
149        .inner_size(480.0, 720.0)
150        .on_navigation(move |url| {
151            let url_str = url.as_str();
152            if url_str.starts_with(AMAZON_REDIRECT_PREFIX) {
153                let code = url
154                    .query_pairs()
155                    .find(|(k, _)| k == "openid.oa2.authorization_code")
156                    .map(|(_, v)| v.to_string());
157                let result = code.ok_or_else(|| {
158                    "Redirect da Amazon alcançado, mas sem 'openid.oa2.authorization_code' na URL"
159                        .to_string()
160                });
161                let _ = tx.send(result);
162                return false;
163            }
164            true
165        })
166        .build()
167        .map_err(|e| {
168            AppError::OAuthConfigError(format!("Falha ao abrir janela de login Amazon: {e}"))
169        })?;
170
171        let code = tokio::task::spawn_blocking(move || {
172            rx.recv_timeout(Duration::from_secs(OAUTH_CALLBACK_TIMEOUT_SECS))
173        })
174        .await
175        .map_err(|e| AppError::OAuthConfigError(format!("Task de callback falhou: {e}")))?
176        .map_err(|_| AppError::OAuthConfigError("Tempo limite de login excedido".to_string()))?
177        .map_err(AppError::OAuthConfigError)?;
178
179        let _ = window.close();
180
181        self.register_device(&code, &client_id_hex, &pkce.verifier, &serial)
182            .await
183    }

185    async fn register_device(
186        &self,
187        code: &str,
188        client_id: &str,
189        verifier: &str,
190        serial: &str,
191    ) -> Result<(), AppError> {
192        let body = serde_json::json!({
193            "auth_data": {
194                "authorization_code": code,
195                "client_domain": "DeviceLegacy",
196                "client_id": client_id,
197                "code_algorithm": "SHA-256",
198                "code_verifier": verifier,
199                "use_global_authentication": false,
200            },
201            "registration_data": {
202                "app_name": AMAZON_APP_NAME,
203                "app_version": AMAZON_APP_VERSION,
204                "device_model": "Windows",
205                "device_name": null,
206                "device_serial": serial,
207                "device_type": AMAZON_DEVICE_TYPE,
208                "domain": "Device",
209                "os_version": "10.0.19044.0",
210            },
211            "requested_extensions": ["customer_info", "device_info"],
212            "requested_token_type": ["bearer", "mac_dms"],
213            "user_context_map": {},
214        });
215
216        let response = HTTP_CLIENT
217            .post(format!("{AMAZON_API}/auth/register"))
218            .header("User-Agent", "AGSLauncher/1.0.0")
219            .json(&body)
220            .send()
221            .await?;
222
223        let status = response.status();
224        let resp_body = response.text().await.unwrap_or_default();
225
226        if !status.is_success() {
227            return Err(AppError::OAuthTokenExchangeError(format!(
228                "Amazon register_device retornou HTTP {status}: {resp_body}"
229            )));
230        }
231
232        let parsed: RegisterDeviceResponse = serde_json::from_str(&resp_body).map_err(|e| {
233            AppError::OAuthTokenExchangeError(format!(
234                "Resposta inesperada do register_device Amazon: {e}"
235            ))
236        })?;
237
238        let success = parsed.response.success;
239        let bearer = success.tokens.bearer;
240
241        let mut extra = HashMap::new();
242        let device_serial = success
243            .extensions
244            .device_info
245            .device_serial_number
246            .unwrap_or_else(|| serial.to_string()); // fallback pro serial gerado localmente
247        extra.insert("device_serial".to_string(), device_serial);
248
249        if let Some(name) = success.extensions.customer_info.given_name {
250            extra.insert("given_name".to_string(), name);
251        }
252
253        let token = OAuthToken {
254            access_token: bearer.access_token,
255            refresh_token: bearer.refresh_token,
256            expires_at: bearer.expires_in.map(|secs| now_unix() + secs as i64),
257            scope: None,
258            extra,
259        };
260
261        save_oauth_token(&self.app_handle, AMAZON_PROVIDER_ID, &token)?;
262
263        Ok(())
264    }
265
266    /// Retorna um access_token válido, renovando via refresh_token se expirado.
267    /// Fluxo de refresh custom (não usa o `refresh_access_token` genérico) porque
268    /// o corpo esperado pela Amazon não segue o formato OAuth2 padrão.
269    pub async fn ensure_valid_token(&self) -> Result<String, AppError> {
270        let stored = load_oauth_token(&self.app_handle, AMAZON_PROVIDER_ID)?
271            .ok_or_else(|| AppError::OAuthTokenNotFound(AMAZON_PROVIDER_ID.to_string()))?;
272
273        if !stored.is_expired() {
274            return Ok(stored.access_token);
275        }
276
277        let refresh_token = stored.refresh_token.clone().ok_or_else(|| {
278            AppError::OAuthTokenNotFound(
279                "amazon: token expirado e sem refresh_token, é necessário novo login".to_string(),
280            )
281        })?;
282
283        let refreshed = refresh_access_token_amazon(&refresh_token).await?;
284
285        let new_token = OAuthToken {
286            access_token: refreshed.access_token,
287            refresh_token: stored.refresh_token.clone(), // Amazon não reemite no refresh
288            expires_at: refreshed.expires_in.map(|secs| now_unix() + secs as i64),
289            scope: stored.scope.clone(),
290            extra: stored.extra.clone(), // preserva device_serial
291        };
292
293        save_oauth_token(&self.app_handle, AMAZON_PROVIDER_ID, &new_token)?;
294
295        Ok(new_token.access_token)
296    }

298    /// Desconecta a conta: tenta desregistrar o dispositivo no servidor da Amazon,
299    /// mas remove o token local mesmo se a chamada falhar (token já expirado, rede fora do ar).
300    pub async fn logout(&self) -> Result<(), AppError> {
301        let Some(stored) = load_oauth_token(&self.app_handle, AMAZON_PROVIDER_ID)? else {
302            return delete_oauth_token(&self.app_handle, AMAZON_PROVIDER_ID);
303        };
304
305        let response = HTTP_CLIENT
306            .post(format!("{AMAZON_API}/auth/deregister"))
307            .header("Authorization", format!("bearer {}", stored.access_token))
308            .header("User-Agent", "AGSLauncher/1.0.0")
309            .json(&serde_json::json!({
310                "request_metadata": {
311                    "app_name": AMAZON_APP_NAME,
312                    "app_version": AMAZON_APP_VERSION,
313                }
314            }))
315            .send()
316            .await;
317
318        if let Ok(resp) = &response {
319            if !resp.status().is_success() {
320                log::warn!(
321                    "Amazon deregister retornou status não-OK; removendo token local mesmo assim"
322                );
323            }
324        }
325
326        delete_oauth_token(&self.app_handle, AMAZON_PROVIDER_ID)
327    }

329    /// Importa a biblioteca completa de entitlements possuídos na conta Amazon.
330    pub async fn fetch_library_detailed(&self) -> Result<Vec<SourceGame>, AppError> {
331        let access_token = self.ensure_valid_token().await?;
332
333        let stored = load_oauth_token(&self.app_handle, AMAZON_PROVIDER_ID)?
334            .ok_or_else(|| AppError::OAuthTokenNotFound(AMAZON_PROVIDER_ID.to_string()))?;
335        let serial = stored.extra.get("device_serial").ok_or_else(|| {
336            AppError::OAuthConfigError(
337                "Amazon: device_serial ausente no token salvo — faça login novamente".to_string(),
338            )
339        })?;
340
341        let hardware_hash = {
342            let mut hasher = Sha256::new();
343            hasher.update(serial.as_bytes());
344            hex_upper(&hasher.finalize())
345        };
346
347        let mut products: HashMap<String, String> = HashMap::new(); // id -> title
348        let mut next_token: Option<String> = None;
349
350        loop {
351            let body = serde_json::json!({
352                "Operation": "GetEntitlements",
353                "clientId": "Sonic",
354                "syncPoint": null,
355                "nextToken": next_token,
356                "maxResults": 50,
357                "productIdFilter": null,
358                "keyId": AMAZON_ENTITLEMENTS_KEY_ID,
359                "hardwareHash": hardware_hash,
360            });
361
362            let response = HTTP_CLIENT
363                .post(AMAZON_GAMING_DISTRIBUTION_ENTITLEMENTS)
364                .header(
365                    "X-Amz-Target",
366                    "com.amazon.animusdistributionservice.entitlement.AnimusEntitlementsService.GetEntitlements",
367                )
368                .header("x-amzn-token", &access_token)
369                // Header não-padrão "UserAgent" (sem hífen) — replicado fielmente do cliente de referência, distinto do header HTTP padrão "User-Agent".
370                .header("UserAgent", "com.amazon.agslauncher.win/3.0.9202.1")
371                .header("Content-Type", "application/json")
372                .header("Content-Encoding", "amz-1.0")
373                .json(&body)
374                .send()
375                .await?;
376
377            let status = response.status();
378            let resp_body = response.text().await.unwrap_or_default();
379
380            if !status.is_success() {
381                return Err(AppError::NetworkError(format!(
382                    "Amazon entitlements retornou HTTP {status}: {resp_body}"
383                )));
384            }
385
386            let parsed: AmazonEntitlementsResponse =
387                serde_json::from_str(&resp_body).map_err(|e| {
388                    AppError::ParseError(format!(
389                        "Falha ao parsear biblioteca Amazon: {e} — corpo: {resp_body}"
390                    ))
391                })?;
392
393            for entitlement in parsed.entitlements {
394                let id = entitlement.product.id;
395                let title = entitlement.product.title.unwrap_or_else(|| id.clone());
396                products.entry(id).or_insert(title);
397            }
398
399            match parsed.next_token {
400                Some(t) if !t.is_empty() => next_token = Some(t),
401                _ => break,
402            }
403        }
404
405        let games = products
406            .into_iter()
407            .map(|(id, title)| SourceGame {
408                platform: "Amazon".to_string(),
409                platform_game_id: id,
410                name: Some(title),
411                installed: false,
412                executable_path: None,
413                install_path: None,
414                playtime_minutes: None,
415                last_played: None,
416            })
417            .collect();
418
419        Ok(games)
420    }

432/// Aceita `expires_in` tanto como número quanto como string (a Amazon retorna como string em `register_device`).
433fn deserialize_expires_in<'de, D>(deserializer: D) -> Result<Option<u64>, D::Error>
434where
435    D: serde::Deserializer<'de>,
436{
437    #[derive(Deserialize)]
438    #[serde(untagged)]
439    enum StringOrNumber {
440        String(String),
441        Number(u64),
442    }
443
444    let opt = Option::<StringOrNumber>::deserialize(deserializer)?;
445    Ok(match opt {
446        Some(StringOrNumber::String(s)) => s.parse::<u64>().ok(),
447        Some(StringOrNumber::Number(n)) => Some(n),
448        None => None,
449    })
450}

452async fn refresh_access_token_amazon(
453    refresh_token: &str,
454) -> Result<AmazonRefreshResponse, AppError> {
455    let body = serde_json::json!({
456        "source_token": refresh_token,
457        "source_token_type": "refresh_token",
458        "requested_token_type": "access_token",
459        "app_name": AMAZON_APP_NAME,
460        "app_version": AMAZON_APP_VERSION,
461    });
462
463    let response = HTTP_CLIENT
464        .post(format!("{AMAZON_API}/auth/token"))
465        .header("User-Agent", "AGSLauncher/1.0.0")
466        .json(&body)
467        .send()
468        .await?;
469
470    let status = response.status();
471    let resp_body = response.text().await.unwrap_or_default();
472
473    if !status.is_success() {
474        return Err(AppError::OAuthTokenExchangeError(format!(
475            "Amazon refresh retornou HTTP {status}: {resp_body}"
476        )));
477    }
478
479    serde_json::from_str(&resp_body).map_err(|e| {
480        AppError::OAuthTokenExchangeError(format!("Resposta inesperada do refresh Amazon: {e}"))
481    })
482}

484fn build_amazon_auth_url(client_id_hex: &str, challenge: &str) -> Result<Url, AppError> {
485    let mut url = Url::parse("https://www.amazon.com/ap/signin")
486        .map_err(|e| AppError::OAuthConfigError(format!("URL de login Amazon inválida: {e}")))?;
487
488    url.query_pairs_mut()
489        .append_pair("openid.ns", "http://specs.openid.net/auth/2.0")
490        .append_pair(
491            "openid.claimed_id",
492            "http://specs.openid.net/auth/2.0/identifier_select",
493        )
494        .append_pair(
495            "openid.identity",
496            "http://specs.openid.net/auth/2.0/identifier_select",
497        )
498        .append_pair("openid.mode", "checkid_setup")
499        .append_pair("openid.oa2.scope", "device_auth_access")
500        .append_pair("openid.ns.oa2", "http://www.amazon.com/ap/ext/oauth/2")
501        .append_pair("openid.oa2.response_type", "code")
502        .append_pair("openid.oa2.code_challenge_method", "S256")
503        .append_pair("openid.oa2.client_id", &format!("device:{client_id_hex}"))
504        .append_pair("language", "en_US")
505        .append_pair("marketPlaceId", AMAZON_MARKETPLACE_ID)
506        .append_pair("openid.return_to", "https://www.amazon.com")
507        .append_pair("openid.pape.max_auth_age", "0")
508        .append_pair("openid.assoc_handle", AMAZON_ASSOC_HANDLE)
509        .append_pair("pageId", AMAZON_ASSOC_HANDLE)
510        .append_pair("openid.oa2.code_challenge", challenge);
511
512    Ok(url)
513}

534#[cfg(target_os = "windows")]
535fn resolve_db_path() -> Option<PathBuf> {
536    let local_appdata = std::env::var("LOCALAPPDATA").ok()?;
537    let path = PathBuf::from(local_appdata)
538        .join("Amazon Games")
539        .join("Data")

551/// Copia o DB pra um arquivo temporário antes de abrir — evita conflito de lock com o Amazon Games App, caso esteja rodando ao mesmo tempo.
552fn copy_to_temp(source: &PathBuf) -> Result<PathBuf, AppError> {
553    let temp_path = std::env::temp_dir().join("playlite_amazon_gameinstallinfo.sqlite");
554    std::fs::copy(source, &temp_path)?;
555    Ok(temp_path)
556}

558/// Importa jogos instalados detectados no banco local do Amazon Games App.
559pub fn import_installed() -> Result<Vec<SourceGame>, AppError> {
560    let Some(db_path) = resolve_db_path() else {
561        return Ok(vec![]); // Amazon Games App não instalado, ou não é Windows
562    };
563
564    let temp_path = copy_to_temp(&db_path)?;
565    let conn = rusqlite::Connection::open_with_flags(
566        &temp_path,
567        rusqlite::OpenFlags::SQLITE_OPEN_READ_ONLY,
568    )?;
569
570    let mut stmt = conn.prepare(
571        "SELECT Id, InstallDirectory, Installed, ProductTitle FROM DbSet WHERE Installed = 1",
572    )?;
573
574    let games = stmt
575        .query_map([], |row| {
576            let id: String = row.get(0)?;
577            let install_dir: Option<String> = row.get(1)?;
578            let title: Option<String> = row.get(3)?;
579
580            Ok(SourceGame {
581                platform: "Amazon".to_string(),
582                platform_game_id: id,
583                name: title,
584                installed: true,
585                executable_path: None, // não disponível nesta tabela
586                install_path: install_dir,
587                playtime_minutes: None,
588                last_played: None,
589            })
590        })?
591        .filter_map(|r| r.ok())
592        .collect();
593
594    let _ = std::fs::remove_file(&temp_path);
595
596    Ok(games)
597}

599/// Cruza a biblioteca completa (OAuth) com os jogos detectados localmente.
600/// Diferente da Epic, o merge é por `platform_game_id` exato (mesmo formato de ID em ambas as fontes),
601/// não por nome — mais confiável, sem falso-positivo/negativo por variação de grafia entre local e API.
602pub fn merge_local_install_status(
603    library_games: &mut Vec<SourceGame>,
604    local_games: Vec<SourceGame>,
605) {
606    for local in local_games {
607        let matched = library_games
608            .iter_mut()
609            .find(|g| g.platform_game_id == local.platform_game_id);
610
611        match matched {
612            Some(g) => {
613                g.installed = true;
614                g.install_path = local.install_path;
615            }
616            None => library_games.push(local), // instalado mas ausente da lib (raro; mantém mesmo assim)
617        }
618    }
