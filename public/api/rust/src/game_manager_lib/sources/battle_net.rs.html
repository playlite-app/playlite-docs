1//! Source para importar jogos instalados via Battle.net (Blizzard/Activision)
2//!
3//! Detecta jogos instalados lendo `product.db`, o arquivo binário (protobuf) que o próprio
4//! Battle.net Agent usa para controlar o que está instalado. Enriquece com `aggregate.json`
5//! quando disponível (nome legível, caminho do executável, último jogado).
6//!
7//! **Por que duas fontes:**
8//! - `product.db` é o arquivo que o Battle.net Agent usa, então é a fonte mais confiável. Não tem nome legível do jogo.
9//! - `aggregate.json` não é documentado oficialmente, é ligado à integração com o app Xbox ("Play Anywhere") pós-aquisição da Microsoft.
10//! - Não há garantia de que `aggregate.json` exista, esteja completo, ou seja mantido no futuro. Por isso é usado para enriquecimento, e não fonte primária.
11//!
12//! **Observações:**
13//! - **Windows:** `C:\ProgramData\Battle.net\Agent\product.db` e `aggregate.json` no mesmo dir.
14//! - **Linux:** não suportado por ora (Battle.net não roda de forma confiável via Wine).
15

98impl BattleNetSource {
99    pub fn new() -> Self {
100        Self::default()
101    }
102
103    /// Resolve o diretório do Battle.net Agent (onde ficam `product.db` e `aggregate.json`).
104    fn resolve_agent_dir(&self) -> Option<PathBuf> {
105        #[cfg(target_os = "windows")]
106        {
107            let path = PathBuf::from(BATTLE_NET_AGENT_DIR_WINDOWS);
108            if path.exists() {
109                return Some(path);
110            }
111        }
112
113        None
114    }
115
116    /// Importa todos os jogos instalados detectados via `product.db`, enriquecidos com `aggregate.json` quando disponível.
117    pub async fn import_installed(&self) -> Result<Vec<SourceGame>, AppError> {
118        let detailed = self.fetch_games_detailed().await?;
119        Ok(detailed.into_iter().map(|game| game.source).collect())
120    }
121
122    /// Importa todos os jogos instalados com metadados extras como capa.
123    pub async fn fetch_games_detailed(&self) -> Result<Vec<BattleNetGame>, AppError> {
124        let Some(agent_dir) = self.resolve_agent_dir() else {
125            return Ok(vec![]); // Battle.net não instalado
126        };
127
128        let product_db_path = agent_dir.join("product.db");
129        if !product_db_path.exists() {
130            return Ok(vec![]); // Agent instalado, mas nenhum jogo ainda
131        }
132
133        let db_bytes = fs::read(&product_db_path)?;
134        let entries = parse_product_db(&db_bytes);
135
136        let aggregate_map = Self::load_aggregate_map(&agent_dir.join("aggregate.json"));
137
138        let mut games = Vec::with_capacity(entries.len());
139
140        for entry in entries {
141            // Pseudo-produtos internos do client (Agent, Desktop App) não são jogos.
142            if BATTLE_NET_INTERNAL_PRODUCT_IDS
143                .iter()
144                .any(|id| entry.product_id.eq_ignore_ascii_case(id))
145            {
146                continue;
147            }
148
149            let Some(install_path) = entry.install_path.as_ref() else {
150                log::warn!(
151                    "product.db: entrada '{}' sem install_path, ignorando",
152                    entry.product_id
153                );
154                continue;
155            };
156
157            // Descarta registros obsoletos (jogo removido manualmente, sem passar pelo Agent).
158            if !Path::new(install_path).is_dir() {
159                log::warn!(
160                    "product.db: pasta de instalação de '{}' não existe mais ({}), ignorando",
161                    entry.product_id,
162                    install_path
163                );
164                continue;
165            }
166
167            let enrichment = aggregate_map.get(&entry.product_id.to_lowercase());
168
169            // Sem tabela ProductId -> Nome própria (como a que o Playnite mantém manualmente), usa o próprio código do produto quando aggregate.json não tem o nome.
170            let name = enrichment
171                .and_then(|g| g.name.clone())
172                .or_else(|| {
173                    BATTLE_NET_CATALOG
174                        .get(&entry.product_id.to_lowercase())
175                        .cloned()
176                })
177                .unwrap_or_else(|| entry.product_id.clone());
178
179            let executable_path = enrichment.and_then(|g| g.icon_path.clone());
180
181            let last_played = enrichment
182                .and_then(|g| g.last_played_timestamp)
183                .filter(|&ms| ms > 0) // 0 = nunca jogado, não um timestamp válido
184                .map(|ms| ms / 1000); // aggregate.json usa ms; SourceGame espera segundos
185
186            let cover_url =
187                enrichment.and_then(|g| g.box_art_uri.clone().or_else(|| g.logo_art_uri.clone()));
188
189            games.push(BattleNetGame {
190                source: SourceGame {
191                    platform: "BattleNet".to_string(),
192                    platform_game_id: entry.product_id,
193                    name: Some(name),
194                    installed: true,
195                    executable_path,
196                    install_path: Some(install_path.clone()),
197                    playtime_minutes: None, // Battle.net não expõe tempo jogado localmente
198                    last_played,
199                },
200                cover_url,
201            });
202        }
203
204        Ok(games)
205    }
206
207    /// Carrega `aggregate.json`, indexado por `product_id` em minúsculas.
208    ///
209    /// Ausência do arquivo (ou erro de parse) não é um erro fatal — só significa importar sem enriquecimento.
210    fn load_aggregate_map(path: &Path) -> HashMap<String, AggregateGame> {
211        let mut map = HashMap::new();
212
213        let content = match fs::read_to_string(path) {
214            Ok(content) => content,
215            Err(_) => {
216                log::info!("aggregate.json não encontrado, seguindo sem enriquecimento");
217                return map;
218            }
219        };
220
221        let parsed: AggregateFile = match serde_json::from_str(&content) {
222            Ok(parsed) => parsed,
223            Err(err) => {
224                log::warn!("Falha ao parsear aggregate.json: {err}");
225                return map;
226            }
227        };
228
229        for game in parsed.installed {
230            if let Some(product_id) = &game.product_id {
231                map.insert(product_id.to_lowercase(), game);
232            }
233        }
234
235        map
236    }
237}

246// === PARSER MANUAL DE PROTOBUF PARA PRODUCT.DB ===
247//
248// Implementação mínima e defensiva: só o suficiente do wire format do protobuf para extrair os três campos.
249// Evita puxar uma dependência de protobuf (prost + build.rs) para um caso de uso tão pequeno.
250
251/// Decodifica todas as entradas de `product.db`.
252///
253/// - O arquivo é uma mensagem raiz com um campo 'repetido de número 1 (`InstalledProductInfo`).
254/// - Cada entrada é lida como qualquer campo length-delimited: tag + length (varint) + conteúdo.
255/// - Uma tag ou length ilegível interrompe o parse (perda de sincronia com o restante).
256/// - Entradas malformadas são puladas com log.
257fn parse_product_db(data: &[u8]) -> Vec<ProductDbEntry> {
258    let mut entries = Vec::new();
259    let mut pos = 0;
260
261    while pos < data.len() {
262        let Some((field_number, wire_type, tag_end)) = read_tag(data, pos) else {
263            log::warn!("product.db: falha ao ler tag em pos {pos}, interrompendo parse");
264            break;
265        };
266        let Some((value, new_pos)) = read_field_value(data, tag_end, wire_type) else {
267            log::warn!(
268                "product.db: falha ao ler valor do campo {field_number} em pos {pos}, interrompendo parse"
269            );
270            break;
271        };
272
273        // Só o campo 1 (repeated InstalledProductInfo) interessa no nível raiz;
274        // Qualquer outro campo já foi consumido/pulado por `read_field_value` acima.
275        if field_number == 1 && wire_type == 2 {
276            if let Some(bytes) = value {
277                match parse_installed_product_info(bytes) {
278                    Some(entry) => entries.push(entry),
279                    None => log::warn!("product.db: entrada malformada ignorada em pos {pos}"),
280                }
281            }
282        }
283
284        pos = new_pos;
285    }
286
287    entries
288}

290/// Decodifica uma mensagem `InstalledProductInfo`:
291///
292/// - Campo 1 = InternalId (string)
293/// - Campo 2 = ProductId (string)
294/// - Campo 3 = Data { campo 1 = Path (string) }
295fn parse_installed_product_info(data: &[u8]) -> Option<ProductDbEntry> {
296    let mut internal_id: Option<String> = None;
297    let mut product_id: Option<String> = None;
298    let mut install_path: Option<String> = None;
299
300    let mut pos = 0;
301    while pos < data.len() {
302        let (field_number, wire_type, tag_end) = read_tag(data, pos)?;
303        let (value, new_pos) = read_field_value(data, tag_end, wire_type)?;
304
305        match (field_number, value) {
306            (1, Some(bytes)) => internal_id = Some(String::from_utf8_lossy(bytes).into_owned()),
307            (2, Some(bytes)) => product_id = Some(String::from_utf8_lossy(bytes).into_owned()),
308            (3, Some(bytes)) => install_path = parse_install_data(bytes),
309            _ => {} // campo desconhecido: valor já foi "consumido" por read_field_value
310        }
311
312        pos = new_pos;
313    }
314
315    Some(ProductDbEntry {
316        internal_id: internal_id.unwrap_or_default(),
317        product_id: product_id?,
318        install_path,
319    })
320}

322/// Decodifica a submensagem `Data`: campo 1 = Path (string).
323fn parse_install_data(data: &[u8]) -> Option<String> {
324    let mut path = None;
325    let mut pos = 0;
326
327    while pos < data.len() {
328        let (field_number, wire_type, tag_end) = read_tag(data, pos)?;
329        let (value, new_pos) = read_field_value(data, tag_end, wire_type)?;
330
331        if field_number == 1 {
332            if let Some(bytes) = value {
333                path = Some(String::from_utf8_lossy(bytes).into_owned());
334            }
335        }
336
337        pos = new_pos;
338    }
339
340    path
341}

343/// Lê a tag de um campo protobuf (varint) e devolve (número_do_campo, wire_type, novo_pos).
344fn read_tag(data: &[u8], pos: usize) -> Option<(u32, u8, usize)> {
345    let (tag, new_pos) = read_varint(data, pos)?;
346    let field_number = (tag >> 3) as u32;
347    let wire_type = (tag & 0x07) as u8;
348    Some((field_number, wire_type, new_pos))
349}

351/// Lê o valor de um campo de acordo com seu wire type, avançando `pos` corretamente mesmo para campos que serão ignorados.
352/// Para wire_type 2 (length-delimited), devolve o slice do valor.
353fn read_field_value(data: &[u8], pos: usize, wire_type: u8) -> Option<(Option<&[u8]>, usize)> {
354    match wire_type {
355        0 => {
356            // varint
357            let (_, new_pos) = read_varint(data, pos)?;
358            Some((None, new_pos))
359        }
360        1 => {
361            // 64-bit fixo
362            let new_pos = pos.checked_add(8)?;
363            (new_pos <= data.len()).then_some((None, new_pos))
364        }
365        2 => {
366            // length-delimited: string, bytes ou submensagem
367            let (len, new_pos) = read_varint(data, pos)?;
368            let len = len as usize;
369            let end = new_pos.checked_add(len)?;
370            (end <= data.len()).then_some((Some(&data[new_pos..end]), end))
371        }
372        5 => {
373            // 32-bit fixo
374            let new_pos = pos.checked_add(4)?;
375            (new_pos <= data.len()).then_some((None, new_pos))
376        }
377        _ => None, // wire type desconhecido: não sabemos avançar com segurança
378    }
379}

381/// Lê um varint (LEB128 base 128) a partir de `pos`. Devolve (valor, novo_pos).
382fn read_varint(data: &[u8], mut pos: usize) -> Option<(u64, usize)> {
383    let mut result: u64 = 0;
384    let mut shift = 0;
385
386    loop {
387        let byte = *data.get(pos)?;
388        pos += 1;
389        result |= ((byte & 0x7F) as u64) << shift;
390        if byte & 0x80 == 0 {
391            return Some((result, pos));
392        }
393        shift += 7;
394        if shift >= 64 {
395            return None; // varint malformado (excede 64 bits)
396        }
397    }
