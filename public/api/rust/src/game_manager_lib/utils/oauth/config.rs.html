47/// Resposta bruta do endpoint de token do provedor.
48#[derive(Debug, Deserialize)]
49pub struct TokenResponse {
50    pub access_token: String,
51    pub refresh_token: Option<String>,
52    #[serde(default)]
53    pub expires_in: Option<u64>, // segundos
54    #[serde(default)]
55    pub token_type: Option<String>,
56    #[serde(default, deserialize_with = "deserialize_scope")]
57    pub scope: Option<String>,
58}

60/// Forma persistida do token (o que vai encriptado pro secrets.db).
61#[derive(Debug, Serialize, Deserialize, Clone)]
62pub struct OAuthToken {
63    pub access_token: String,
64    pub refresh_token: Option<String>,
65    pub expires_at: Option<i64>, // timestamp unix; None = expiração desconhecida
66    pub scope: Option<String>,
67}
68
69impl From<TokenResponse> for OAuthToken {
70    fn from(resp: TokenResponse) -> Self {
71        let expires_at = resp.expires_in.map(|secs| now_unix() + secs as i64);
72        Self {
73            access_token: resp.access_token,
74            refresh_token: resp.refresh_token,
75            expires_at,
76            scope: resp.scope,
77        }
78    }
79}
80
81impl OAuthToken {
82    /// Considera expirado 60s antes do prazo real, pra dar margem de segurança contra latência de rede durante uma chamada de API.
83    pub fn is_expired(&self) -> bool {
84        match self.expires_at {
85            Some(exp) => now_unix() >= exp - 60,
86            None => false,
87        }
88    }
89}

93fn now_unix() -> i64 {
94    SystemTime::now()
95        .duration_since(UNIX_EPOCH)
96        .unwrap_or_default()
97        .as_secs() as i64
98}

100/// Alguns provedores (GOG) retornam `scope` como string; outros (Epic) retornam
101/// como array de strings, inclusive vazio (`[]`) quando não há escopos.
102/// Normaliza os dois formatos: array vira string com join por espaço, array vazio vira `None`.
103fn deserialize_scope<'de, D>(deserializer: D) -> Result<Option<String>, D::Error>
104where
105    D: serde::Deserializer<'de>,
106{
107    #[derive(Deserialize)]
108    #[serde(untagged)]
109    enum ScopeField {
110        Text(String),
111        List(Vec<String>),
112    }
113
114    let opt = Option::<ScopeField>::deserialize(deserializer)?;
115    Ok(match opt {
116        Some(ScopeField::Text(s)) => Some(s),
117        Some(ScopeField::List(list)) if list.is_empty() => None,
118        Some(ScopeField::List(list)) => Some(list.join(" ")),
119        None => None,
120    })
121}

123async fn parse_token_response(
124    response: reqwest::Response,
125    provider_id: &str,
126    stage: &str,
127) -> Result<TokenResponse, AppError> {
128    let status = response.status();
129    let body = response.text().await.unwrap_or_default();
130
131    if !status.is_success() {
132        return Err(AppError::OAuthTokenExchangeError(format!(
133            "[{provider_id}] falha em '{stage}': HTTP {status} - {body}"
134        )));
135    }
136
137    serde_json::from_str::<TokenResponse>(&body).map_err(|e| {
138        AppError::OAuthTokenExchangeError(format!(
139            "[{provider_id}] resposta inesperada em '{stage}': {e}"
140        ))
141    })
142}

144/// Monta a requisição do endpoint de token, aplicando client_id/secret no corpo
145/// ou via header `Authorization: Basic`, conforme `token_auth_method` do provedor.
146fn build_token_request(
147    config: &OAuthProviderConfig,
148    mut params: Vec<(&'static str, String)>,
149) -> reqwest::RequestBuilder {
150    let mut request = match config.token_request_method {
151        TokenRequestMethod::Get => HTTP_CLIENT.get(&config.token_endpoint),
152        TokenRequestMethod::Post => HTTP_CLIENT.post(&config.token_endpoint),
153    };
154
155    match config.token_auth_method {
156        TokenAuthMethod::Body => {
157            params.push(("client_id", config.client_id.clone()));
158            if let Some(secret) = &config.client_secret {
159                params.push(("client_secret", secret.clone()));
160            }
161        }
162        TokenAuthMethod::BasicHeader => {
163            let secret = config.client_secret.clone().unwrap_or_default();
164            request = request.basic_auth(&config.client_id, Some(secret));
165        }
166    }
167
168    match config.token_request_method {
169        TokenRequestMethod::Get => request.query(&params),
170        TokenRequestMethod::Post => request.form(&params),
171    }
172}

176/// Monta a URL de autorização para abrir no navegador do sistema.
177pub fn build_authorize_url(
178    config: &OAuthProviderConfig,
179    pkce: Option<&PkceChallenge>,
180    state: &str,
181) -> Result<Url, AppError> {
182    let mut url = Url::parse(&config.authorize_endpoint)
183        .map_err(|e| AppError::OAuthConfigError(format!("URL de autorização inválida: {e}")))?;
184
185    {
186        let mut query = url.query_pairs_mut();
187        query
188            .append_pair("client_id", &config.client_id)
189            .append_pair("redirect_uri", &config.redirect_uri)
190            .append_pair("response_type", "code")
191            .append_pair("state", state);
192
193        if !config.scopes.is_empty() {
194            query.append_pair("scope", &config.scopes.join(" "));
195        }
196
197        if config.uses_pkce {
198            let pkce = pkce.ok_or_else(|| {
199                AppError::OAuthConfigError(
200                    "uses_pkce=true mas nenhum PkceChallenge foi fornecido".into(),
201                )
202            })?;
203            query
204                .append_pair("code_challenge", &pkce.challenge)
205                .append_pair("code_challenge_method", "S256");
206        }
207
208        for (key, value) in &config.extra_params {
209            query.append_pair(key, value);
210        }
211    }
212
213    Ok(url)
214}

216/// Troca o `code` recebido no callback por um access_token.
217pub async fn exchange_code_for_token(
218    config: &OAuthProviderConfig,
219    code: &str,
220    pkce_verifier: Option<&str>,
221) -> Result<TokenResponse, AppError> {
222    let mut params = vec![
223        ("grant_type", "authorization_code".to_string()),
224        ("code", code.to_string()),
225        ("redirect_uri", config.redirect_uri.clone()),
226    ];
227    if let Some(verifier) = pkce_verifier {
228        params.push(("code_verifier", verifier.to_string()));
229    }
230
231    let response = build_token_request(config, params).send().await?;
232    parse_token_response(response, config.provider_id, "exchange").await
233}

235/// Renova o access_token usando o refresh_token salvo.
236pub async fn refresh_access_token(
237    config: &OAuthProviderConfig,
238    refresh_token: &str,
239) -> Result<TokenResponse, AppError> {
240    let params = vec![
241        ("grant_type", "refresh_token".to_string()),
242        ("refresh_token", refresh_token.to_string()),
243    ];
244
245    let response = build_token_request(config, params).send().await?;
246    parse_token_response(response, config.provider_id, "refresh").await
247}
