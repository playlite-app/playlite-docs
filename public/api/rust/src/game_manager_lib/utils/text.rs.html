5const EDITION_SUFFIXES: &[&str] = &[

7    "Collectors Edition",

11    "Special Edition",

20/// Remove símbolos de marca registrada, preservando capitalização e conteúdo.
21/// Uso: limpar nome antes de persistir (import), exibição.
22pub fn strip_trademark_symbols(name: &str) -> String {
23    name.chars()
24        .filter(|c| !matches!(c, '™' | '®' | '©'))
25        .collect::<String>()
26        .split_whitespace()
27        .collect::<Vec<_>>()
28        .join(" ")
29}
30
31/// Normaliza para comparação/matching: minúsculo, sem símbolos, sem pontuação de separador.
32/// Uso: cache keys, dedup, series matching, comparação com resultado de busca externa.
33pub fn normalize_for_matching(name: &str) -> String {
34    strip_trademark_symbols(name)
35        .to_lowercase()
36        .replace(':', "")
37        .trim()
38        .to_string()
39}

41/// Heurística por palavra-chave para identificar nomes que provavelmente
42/// não são o jogo-base (DLC, edição especial, trilha sonora, demo, pré-venda).
43/// Não é definitivo — é um filtro barato para descartar candidatos óbvios
44/// antes de aplicar critérios mais confiáveis (match exato, dados estruturados).
45pub fn is_likely_non_base_game(name: &str) -> bool {
46    let lower = name.to_lowercase();
47    let keywords = [
48        "season pass",
49        " dlc",
50        "demo",
51        "add-on",
52        "addon",
53        "expansion",
54        "pre-order",
55        "preorder",
56        "pre order",
57        "starter pack",
58        " pack",
59        "soundtrack",
60        "artbook",
61        "art book",
62        "playtest",
63        "goodie",
64    ];
65    keywords.iter().any(|kw| lower.contains(kw))
66}

68/// Remove sufixo de edição conhecido do final do nome, se houver.
69/// Usado como segunda tentativa de match quando o nome completo (com edição) não bate.
70pub fn strip_edition_suffix(name: &str) -> String {
71    let trimmed = name.trim();
72    for suffix in EDITION_SUFFIXES {
73        if let Some(rest) = trimmed.to_lowercase().strip_suffix(&suffix.to_lowercase()) {
74            return trimmed[..rest.len()]
75                .trim_end_matches(['-', ':', ',', ' '])
76                .trim()
77                .to_string();
78        }
79    }
80    trimmed.to_string()
81}
