1//! Sistema de migração de schema do banco de dados.
2//!
3//! Regras:
4//! - Migrações só ocorrem entre MAJOR versions
5//! - Sempre assume backup automático antes de migrar
6//! - Falha de migração nunca destrói dados existentes
7//!
8//! **Nota:** Durante a fase de testes, as migrações estão desabilitadas.
9//! O banco deve ser deletado e recriado manualmente para aplicar mudanças de schema.
10
11use crate::database::{current_schema_version, expected_schema_version};
12use crate::errors::AppError;
13use rusqlite::Connection;
14use tauri::AppHandle;
15
16/// Executa migrações de schema se necessário.
17///
18/// Deve ser chamada logo após abrir o banco e ANTES de usar qualquer tabela.
19pub fn run_migrations(app: &AppHandle, conn: &Connection) -> Result<(), AppError> {
20    let current = current_schema_version(conn)?;
21    let expected = expected_schema_version(app);
22
23    // Banco novo (schema já foi criado em initialize_databases)
24    if current == 0 {
25        return Ok(());
26    }
27
28    // Nada a fazer
29    if current == expected {
30        return Ok(());
31    }
32
33    // Downgrade não suportado
34    if current > expected {
35        return Err(AppError::ValidationError(format!(
36            "Downgrade não suportado. Banco v{}, app espera v{}",
37            current, expected
38        )));
39    }
40
41    // Durante fase de testes, migrações estão desabilitadas
42    // O usuário deve deletar e recriar o banco manualmente
43    Err(AppError::ValidationError(format!(
44        "Migração necessária de v{} para v{}. Durante a fase de testes, delete o banco e deixe o app recriá-lo.",
45        current, expected
46    )))
47}
