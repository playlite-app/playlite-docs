5/// Resultado simplificado usado pelo frontend.
6/// Estrutura compatível com: Wishlist, Busca, Descoberta e Similares
7#[derive(Debug, Clone, Serialize, Deserialize)]
8pub struct GameBrainSearchResult {
9    pub id: String,
10    pub name: String,
11    pub cover_url: Option<String>,
12    pub genre: Option<String>,
13    pub year: Option<u32>,
14    pub rating: Option<f64>,
15    pub platforms: Vec<String>,
16    pub link: Option<String>,
17}
18
19/// Todos os campos são opcionais — use apenas o que precisar.
20///
21/// # Exemplo
22///
23/// ```rust
24/// let params = GameBrainSearchParams {
25///     filters: vec![
26///         GameBrainFilter {
27///             key: "platform".into(),
28///             values: vec![GameBrainFilterValue { value: "pc".into() }],
29///             connection: Some("OR".into()),
30///         },
31///     ],
32///     sort: Some(GameBrainSort::Rating),
33///     sort_order: Some(GameBrainSortOrder::Desc),
34///     limit: Some(20),
35///     offset: None,
36/// };
37/// ```
38#[derive(Debug, Default, Clone)]
39pub struct GameBrainSearchParams {
40    /// Filtros a aplicar na busca. Os valores válidos vêm do campo `filter_options` da resposta da API.
41    pub filters: Vec<GameBrainFilter>,
42    pub sort: Option<GameBrainSort>,
43    /// Direção da ordenação. Padrão da API: descendente.
44    pub sort_order: Option<GameBrainSortOrder>,
45    /// Número máximo de resultados. Padrão da API: 10.
46    pub limit: Option<u32>,
47    pub offset: Option<u32>,
48}
49
50/// Um filtro a ser aplicado na busca.
51#[derive(Debug, Clone, Serialize)]
52pub struct GameBrainFilter {
53    /// Chave do filtro (ex: "platform", "genre", "play_mode").
54    pub key: String,
55    pub values: Vec<GameBrainFilterValue>,
56    /// Operador lógico entre os valores: "OR" ou "AND". A maioria dos filtros usa "OR". Omita para usar o padrão da API.
57    #[serde(skip_serializing_if = "Option::is_none")]
58    pub connection: Option<String>,
59}
60
61/// Um valor individual de filtro (ex: "pc", "action", "single_player").
62#[derive(Debug, Clone, Serialize)]
63pub struct GameBrainFilterValue {
64    pub value: String,
65}

67#[derive(Debug, Clone)]
68pub enum GameBrainSort {
69    Rating,
70    ReleaseDate,
71    Price,
72}
73
74impl GameBrainSort {
75    pub(crate) fn as_str(&self) -> &'static str {
76        match self {
77            GameBrainSort::Rating => "computed_rating",
78            GameBrainSort::ReleaseDate => "release_date",
79            GameBrainSort::Price => "price",
80        }
81    }
82}
83
84#[derive(Debug, Clone)]
85pub enum GameBrainSortOrder {
86    Asc,
87    Desc,
88}
89
90impl GameBrainSortOrder {
91    pub(crate) fn as_str(&self) -> &'static str {
92        match self {
93            GameBrainSortOrder::Asc => "asc",
94            GameBrainSortOrder::Desc => "desc",
95        }
96    }
97}

99/// Jogo similar retornado para o frontend.
100///
101/// Subset dos campos disponíveis na resposta de /similar.
102/// Inclui screenshots e micro_trailer para a UI da aba Descoberta.
103#[derive(Debug, Clone, Serialize, Deserialize)]
104#[serde(rename_all = "camelCase")]
105pub struct SimilarGame {
106    /// ID no formato "gamebrain:{id}" para consistência com GameBrainSearchResult.
107    pub id: String,
108    pub name: String,
109    pub cover_url: Option<String>,
110    pub genre: Option<String>,
111    pub year: Option<u32>,
112    /// Rating em percentual (0–100), convertido de 0.0–1.0.
113    pub rating: Option<f64>,
114    pub link: Option<String>,
115    /// Primeiros screenshots disponíveis (máx 4 pela API).
116    pub screenshots: Vec<String>,
117    /// URL do micro-trailer em .webm (Steam CDN), se disponível.
118    pub micro_trailer: Option<String>,
119    /// Flag de conteúdo adulto — o frontend pode usar para bloquear a exibição automática de imagens.
120    pub adult_only: bool,
121}

123/// Mídia de um jogo retornada para a aba Mídia do GameWindow.
124///
125/// Separa os vídeos em dois grupos já classificados:
126/// - `trailers`: `.webm` do Steam CDN — reprodução direta no `<video>`
127/// - `youtube_embeds`: URLs `youtube-nocookie.com` — renderizadas em `<iframe>`
128///
129/// Essa separação acontece no backend para não vazar lógica de parsing de URL para o frontend.
130#[derive(Debug, Clone, Serialize, Deserialize)]
131pub struct GameMedia {
132    pub screenshots: Vec<String>,
133    /// `.webm` diretos (Steam CDN). Prontos para `<video src=...>`.
134    pub trailers: Vec<String>,
135    /// Embeds do YouTube (`youtube-nocookie.com/embed/...`). Prontos para `<iframe src=...>`.
136    pub youtube_embeds: Vec<String>,
137    /// Micro-trailer de loop curto, se disponível.
138    pub micro_trailer: Option<String>,
139}
