12#[tauri::command]
13pub async fn scan_games_folder(folder_path: String) -> Result<ScanResult, String> {
14    let path = Path::new(&folder_path);
15
16    // Validações básicas
17    if !path.exists() {
18        return Ok(ScanResult {
19            success: false,
20            message: "Pasta não encontrada".to_string(),
21            discoveries: vec![],
22        });
23    }
24
25    if !path.is_dir() {
26        return Ok(ScanResult {
27            success: false,
28            message: "Caminho não é uma pasta".to_string(),
29            discoveries: vec![],
30        });
31    }
32
33    // Executar scan
34    let discoveries = scan_folder(path)?;
35
36    let message = if discoveries.is_empty() {
37        "Nenhum jogo encontrado nesta pasta".to_string()
38    } else {
39        format!("Encontrados {} possíveis jogos", discoveries.len())
40    };
41
42    Ok(ScanResult {
43        success: true,
44        message,
45        discoveries,
46    })
47}

49/// Adiciona um jogo descoberto pelo scan ao banco de dados.
50#[tauri::command]
51pub async fn add_game_from_scan(
52    app: AppHandle,
53    state: State<'_, AppState>,
54    name: String,
55    executable_path: String,
56    base_path: String,
57) -> Result<String, AppError> {
58    use crate::sources::providers::SourceGame;
59
60    let game = SourceGame {
61        platform: "Outra".to_string(),
62        platform_game_id: executable_path.clone(),
63        name: Some(name),
64        installed: true,
65        executable_path: Some(executable_path.clone()),
66        install_path: Some(base_path),
67        playtime_minutes: Some(0),
68        last_played: None,
69    };
70
71    let (inserted, _) = persist_source_games(&state, vec![game]).await?;
72
73    if inserted == 0 {
74        return Err(AppError::ValidationError(
75            "Este jogo já foi adicionado anteriormente.".to_string(),
76        ));
77    }
78
79    let _ = app.emit("library_updated", ());
80
81    Ok("Jogo adicionado com sucesso.".to_string())
82}

84/// Adiciona múltiplos jogos descobertos pelo scan ao banco de dados.
85#[tauri::command]
86pub async fn add_games_from_scan(
87    app: AppHandle,
88    state: State<'_, AppState>,
89    games: Vec<ScanGameInput>,
90) -> Result<String, AppError> {
91    use crate::sources::providers::SourceGame;
92
93    let source_games: Vec<SourceGame> = games
94        .into_iter()
95        .map(|g| SourceGame {
96            platform: "Outra".to_string(),
97            platform_game_id: g.executable_path.clone(),
98            name: Some(g.name),
99            installed: true,
100            executable_path: Some(g.executable_path.clone()),
101            install_path: Some(g.base_path),
102            playtime_minutes: Some(0),
103            last_played: None,
104        })
105        .collect();
106
107    let (inserted, updated) = persist_source_games(&state, source_games).await?;
108    let _ = app.emit("library_updated", ());
109
110    Ok(format!("{} adicionados, {} atualizados", inserted, updated))
111}
