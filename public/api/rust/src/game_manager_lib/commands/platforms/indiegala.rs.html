13/// Persiste jogos da IndieGala nas tabelas `games` e `game_details`.
14///
15/// Difere de `persist_source_games` por também gravar `description_raw` e `tags`
16/// em `game_details`. Diferente da Legacy Games, aqui não há `cover_url`.
17///
18/// `playtime_minutes` é passado como `Option` (não `unwrap_or(0)`) pro `UPDATE`
19/// porque no modo `full`, jogos possuídos mas que não foram instalados não têm playtime
20/// conhecido — usar `COALESCE` preserva o valor real já salvo de uma importação anterior
21/// (ex: jogo que foi desinstalado depois de já ter sido jogado) em vez de zerar por engano.
22async fn persist_indiegala_games(
23    state: &AppState,
24    games: Vec<crate::sources::indiegala::IndiegalaGame>,
25) -> Result<(u32, u32), AppError> {
26    let mut conn = state.games_db.lock().map_err(|_| AppError::MutexError)?;
27    let tx = conn
28        .transaction()
29        .map_err(|e| AppError::DatabaseError(e.to_string()))?;
30
31    let mut inserted = 0u32;
32    let mut updated = 0u32;
33    let now = Utc::now().to_rfc3339();
34
35    for indiegala_game in games {
36        let game = &indiegala_game.source;
37
38        let exists: bool = tx
39            .query_row(
40                "SELECT EXISTS(SELECT 1 FROM games WHERE platform = ?1 AND platform_game_id = ?2)",
41                params![&game.platform, &game.platform_game_id],
42                |row| row.get(0),
43            )
44            .unwrap_or(false);
45
46        let status = status_logic::calculate_status(game.playtime_minutes.unwrap_or(0) as i32);
47
48        if !exists {
49            let new_id = Uuid::new_v4().to_string();
50
51            tx.execute(
52                "INSERT INTO games (
53                    id, name, cover_url, platform, platform_game_id,
54                    installed, status, playtime, last_played, added_at,
55                    favorite, user_rating, install_path, executable_path
56                ) VALUES (?1, ?2, NULL, ?3, ?4, ?5, ?6, ?7, NULL, ?8, 0, NULL, ?9, ?10)",
57                params![
58                    new_id,
59                    game.name.as_deref().unwrap_or("Unknown"),
60                    game.platform,
61                    game.platform_game_id,
62                    game.installed,
63                    status,
64                    game.playtime_minutes.unwrap_or(0),
65                    now,
66                    game.install_path,
67                    game.executable_path,
68                ],
69            )
70            .map_err(|e| AppError::DatabaseError(e.to_string()))?;
71
72            let tags_json = indiegala_game
73                .tags
74                .as_ref()
75                .and_then(|tags| crate::database::serialize_tags(tags).ok());
76
77            if indiegala_game.description_raw.is_some() || tags_json.is_some() {
78                tx.execute(
79                    "INSERT OR IGNORE INTO game_details (game_id, description_raw, tags)
80                     VALUES (?1, ?2, ?3)",
81                    params![new_id, indiegala_game.description_raw, tags_json],
82                )
83                .map_err(|e| AppError::DatabaseError(e.to_string()))?;
84            }
85
86            inserted += 1;
87        } else {
88            tx.execute(
89                "UPDATE games SET
90                    installed       = ?1,
91                    status          = ?2,
92                    playtime        = COALESCE(?3, playtime),
93                    install_path    = COALESCE(?4, install_path),
94                    executable_path = COALESCE(?5, executable_path)
95                 WHERE platform = ?6 AND platform_game_id = ?7",
96                params![
97                    game.installed,
98                    status,
99                    game.playtime_minutes, // Option<u32> cru — sem unwrap_or(0)
100                    game.install_path,
101                    game.executable_path,
102                    game.platform,
103                    game.platform_game_id,
104                ],
105            )
106            .map_err(|e| AppError::DatabaseError(e.to_string()))?;
107
108            updated += 1;
109        }
110    }
111
112    tx.commit()
113        .map_err(|e| AppError::DatabaseError(e.to_string()))?;
114
115    Ok((inserted, updated))
116}

118/// Importa jogos da IndieGala via IGClient.
119///
120/// `full=false` (padrão): só jogos instalados no momento, via `installed.json`.
121/// `full=true`: biblioteca completa de posse via `config.json` (`user_collection`),
122/// cruzada com `installed.json` pra marcar o que está instalado e reaproveitar metadados desses casos.
123///
124/// `installed_json_path`/`config_json_path` — caminhos customizados opcionais.
125/// Se omitidos, usam os caminhos padrão do Windows:
126/// - `installed.json`: `%APPDATA%\IGClient\storage\installed.json`
127/// - `config.json`: `%APPDATA%\IGClient\config.json`
128#[tauri::command]
129pub async fn import_indiegala_games(
130    app: AppHandle,
131    state: State<'_, AppState>,
132    full: bool,
133    installed_json_path: Option<String>,
134    config_json_path: Option<String>,
135) -> Result<String, AppError> {
136    let installed_path = installed_json_path
137        .filter(|s| !s.trim().is_empty())
138        .map(std::path::PathBuf::from);
139    let config_path = config_json_path
140        .filter(|s| !s.trim().is_empty())
141        .map(std::path::PathBuf::from);
142
143    let source = IndiegalaSource::new(installed_path);
144
145    let games = if full {
146        source.fetch_full_library_detailed(config_path).await?
147    } else {
148        source.fetch_installed_detailed().await?
149    };
150
151    if games.is_empty() {
152        let msg = if full {
153            "Nenhum jogo IndieGala encontrado na biblioteca."
154        } else {
155            "Nenhum jogo IndieGala instalado encontrado."
156        };
157        return Ok(msg.to_string());
158    }
159
160    let (inserted, updated) = persist_indiegala_games(&state, games).await?;
161    let message = format!(
162        "IndieGala: {} adicionados, {} atualizados",
163        inserted, updated
164    );
165    info!("{}", message);
166
167    let _ = app.emit("library_updated", ());
168
169    Ok(message)
170}
