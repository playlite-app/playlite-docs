30#[cfg_attr(mobile, tauri::mobile_entry_point)]
31pub fn run() {
32    tauri::Builder::default()
33        .plugin(tauri_plugin_shell::init())
34        .plugin(tauri_plugin_opener::init())
35        .plugin(tauri_plugin_store::Builder::default().build())
36        .plugin(tauri_plugin_dialog::init())
37        .plugin(tauri_plugin_updater::Builder::new().build())
38        .setup(|app| {
39            let app_handle = app.handle();
40
41            // === WAYLAND: Define app_id para ícone correto na barra de tarefas ===
42            #[cfg(target_os = "linux")]
43            {
44                glib::set_prgname(Some("Playlite"));
45                glib::set_application_name("Playlite");
46            }
47
48            // === LOGGING ===
49
50            let log_dir = app_handle
51                .path()
52                .app_log_dir()
53                .expect("Falha ao pegar pasta de log");
54
55            std::fs::create_dir_all(&log_dir).expect("Falha ao criar pasta de dev_logs");
56
57            let _guard = logger::init_logging(log_dir.clone());
58            app.manage(_guard);
59
60            // === SEGURANÇA ===
61
62            security::init_security(app_handle).expect("Falha ao inicializar sistema de segurança");
63
64            // === BANCOS DE DADOS ===
65
66            let db_state = database::initialize_databases(app_handle)
67                .expect("Falha ao inicializar bancos de dados");
68
69            app.manage(db_state);
70
71            // === INICIALIZAÇÃO PÓS-UPDATE ===
72
73            // Verifica se houve atualização e faz backup/migração se necessário
74            if let Err(e) = initialize_app(app_handle) {
75                tracing::error!("Erro na inicialização pós-update: {}", e);
76                // Não falha a aplicação, apenas loga o erro
77            }
78
79            // === COLLABORATIVE FILTERING ===
80
81            if let Err(e) = services::cf_aggregator::init_cf_index() {
82                tracing::warn!("CF desativado (fallback CB ativo): {}", e);
83            }
84
85            // Log de inicialização completa
86            tracing::info!("Playlite iniciado com sucesso");
87
88            Ok(())
89        })
90        .invoke_handler(tauri::generate_handler![
91            // Comando de Inicialização do Banco de Dados
92            database::init_db,
93            // Comandos de Backup e Restauração
94            database::backup::export_database,
95            database::backup::import_database,
96            // Comandos de Jogos (CRUD)
97            commands::games::add_game,
98            commands::games::get_games,
99            commands::games::get_library_game_details,
100            commands::games::toggle_favorite,
101            commands::games::delete_game,
102            commands::games::update_game,
103            commands::games::update_game_details,
104            // Comandos da Lista de Desejos
105            commands::wishlist::search_wishlist_game,
106            commands::wishlist::search_wishlist_game_by_features,
107            commands::wishlist::add_to_wishlist,
108            commands::wishlist::get_wishlist,
109            commands::wishlist::remove_from_wishlist,
110            commands::wishlist::check_wishlist_status,
111            commands::wishlist::refresh_prices,
112            commands::wishlist::import_wishlist,
113            commands::wishlist::fetch_wishlist_covers,
114            // Comandos de Importação de Plataformas
115            commands::plataforms::scanner::scan_games_folder,
116            commands::plataforms::scanner::add_game_from_scan,
117            commands::plataforms::scanner::add_games_from_scan,
118            commands::plataforms::steam::import_steam_library,
119            commands::plataforms::epic::import_epic_games,
120            commands::plataforms::heroic::import_heroic_games,
121            commands::plataforms::ubisoft::import_ubisoft_games,
122            commands::plataforms::legacy::import_legacy_games,
123            commands::plataforms::gog::gog_login,
124            commands::plataforms::gog::gog_logout,
125            commands::plataforms::gog::gog_is_authenticated,
126            commands::plataforms::gog::import_gog_games,
127            commands::plataforms::battle_net::import_battle_net_games,
128            // Comandos de Metadados (Enriquecimento, Capas, Refresh, Busca)
129            commands::metadata::enrichment::update_metadata,
130            commands::metadata::get_metadata::fill_missing_metadata,
131            commands::metadata::covers::fetch_missing_covers,
132            commands::metadata::refresh::check_and_refresh_background,
133            commands::metadata::search::get_trending_games,
134            commands::metadata::search::get_upcoming_games,
135            commands::metadata::search::fetch_game_details,
136            commands::metadata::search::get_active_giveaways,
137            commands::metadata::search::get_similar_games,
138            commands::metadata::search::get_game_media,
139            commands::metadata::search::get_profile_similar_games,
140            commands::metadata::pcgamingwiki::get_or_fetch_pcgw_data,
141            commands::metadata::pcgamingwiki::refresh_pcgw_data,
142            commands::metadata::pcgamingwiki::search_pcgw_games,
143            commands::metadata::pcgamingwiki::get_pcgw_scraped_data,
144            // Comandos de Configuração (Secrets)
145            commands::settings::set_secret,
146            commands::settings::get_secret,
147            commands::settings::delete_secret,
148            commands::settings::list_secrets,
149            commands::settings::get_secrets,
150            commands::settings::set_secrets,
151            // Comandos de Recomendação
152            commands::recommendation::core::get_user_profile,
153            commands::recommendation::core::recommend_hybrid_library,
154            commands::recommendation::core::recommend_collaborative_library,
155            commands::recommendation::core::recommend_from_library,
156            commands::recommendation::analysis::generate_recommendation_analysis,
157            // Comandos de Tradução de Descrição
158            commands::ai_translation::translate_description,
159            // Comandos de Conquistas de Jogos
160            commands::achievements::get_recent_achievements,
161            // Comandos de Cache de Metadados
162            commands::caches::cleanup_cache,
163            commands::caches::clear_all_cache,
164            commands::caches::get_detailed_cache_stats,
165            // Comandos de Sistema
166            commands::system::open_folder,
167            commands::system::open_file,
168            // Comandos de Versão e SO
169            commands::version::get_app_version,
170            commands::version::get_app_version_info,
171            commands::version::is_updater_enabled,
172            // Comandos de Imagem
173            services::images::cache_cover_image,
174            services::images::check_local_cover,
175            services::images::clear_cover_cache,
176            // Comandos de Subscricoes
177            commands::subscriptions::get_amazon_luna_catalog,
178            commands::subscriptions::get_game_pass_catalog,
179            commands::subscriptions::get_ea_play_catalog,
180            commands::subscriptions::get_ubisoft_plus_catalog,
181            commands::subscriptions::get_subscription_settings,
182            commands::subscriptions::save_subscription_settings
183        ])
184        .run(tauri::generate_context!())
185        .expect("error while running tauri application");
